# 느낀점
3-4 예제를 `SRP` 와 `OCP` 를 준수하면서 `Switch` 문을 사용하지 말아보자.
```java
public interface PayStrategy {
    Money calculatePay();
}

public class Employee {

    private final PayStrategy payStrategy;

    public Employee(final PayStrategy payStrategy) {
        this.payStrategy = payStrategy;
    }

    public Money calculatePay() {
        return payStrategy.calculatePay();
    }

}

public class CommissionedPayStrategy implements PayStrategy {

    @Override
    public Money calculatePay() {
        // pay logic
    }

}

public class HourlyPayStrategy implements PayStrategy {

    @Override
    public Money calculatePay() {
        // pay logic 
    }

}

public class SalariedPayStrategy implements PayStrategy {

    @Override
    public Money calculatePay() {
        // pay logic
    }

}

public Money calculatePay(Employee employee) {
    return employee.calculatePay();
}
```
